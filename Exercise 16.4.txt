SELECT I.DESCRIPTION, IL.NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

//2
SELECT I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

//3
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) POSTS_VALUE
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME;
